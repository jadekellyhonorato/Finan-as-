    <script>
        // Array para armazenar os dados (Num app real, viria de um banco de dados ou LocalStorage)
        let transacoes = [
            // Exemplo baseado na sua planilha de Fevereiro
            { id: 1, tipo: 'entrada', nome: 'Salário', categoria: 'Salário', valor: 1516.36, dataVencimento: '2026-02-05', pago: true },
            { id: 2, tipo: 'entrada', nome: 'Vale Alimentação', categoria: 'Salário', valor: 71.91, dataVencimento: '2026-02-05', pago: true },
            { id: 3, tipo: 'saida', nome: 'Mãe', categoria: 'Contas', valor: 500.00, dataVencimento: '2026-02-06', pago: false },
            { id: 4, tipo: 'saida', nome: 'Transporte', categoria: 'Transporte', valor: 200.00, dataVencimento: '', pago: false }
        ];

        const form = document.getElementById('transactionForm');
        
        // Define mês atual no cabeçalho
        const meses = ["Janeiro", "Fevereiro", "Março", "Abril", "Maio", "Junho", "Julho", "Agosto", "Setembro", "Outubro", "Novembro", "Dezembro"];
        const hoje = new Date();
        document.getElementById('currentMonth').innerText = `${meses[hoje.getMonth()]} / ${hoje.getFullYear()}`;

        form.addEventListener('submit', function(e) {
            e.preventDefault();
            
            const transacao = {
                id: Date.now(),
                tipo: document.getElementById('tipo').value,
                nome: document.getElementById('nome').value,
                categoria: document.getElementById('categoria').value,
                valor: parseFloat(document.getElementById('valor').value),
                dataVencimento: document.getElementById('dataVencimento').value,
                pago: document.getElementById('pago').checked
            };

            transacoes.push(transacao);
            form.reset();
            atualizarInterface();
        });

        function formatarMoeda(valor) {
            return valor.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' });
        }

        function formatarData(dataIso) {
            if (!dataIso) return '-';
            const partes = dataIso.split('-');
            return `${partes[2]}/${partes[1]}/${partes[0]}`;
        }

        function alternarStatusPagamento(id) {
            const index = transacoes.findIndex(t => t.id === id);
            if (index !== -1) {
                transacoes[index].pago = !transacoes[index].pago;
                atualizarInterface();
            }
        }

        function excluirTransacao(id) {
            transacoes = transacoes.filter(t => t.id !== id);
            atualizarInterface();
        }

        function atualizarInterface() {
            let totalEntradas = 0;
            let totalSaidas = 0;
            let gastosPorCategoria = {};

            const tbody = document.getElementById('tabelaTransacoes');
            tbody.innerHTML = '';

            // Ordena para os mais recentes/não pagos ficarem visíveis
            transacoes.sort((a, b) => a.pago - b.pago || new Date(a.dataVencimento) - new Date(b.dataVencimento));

            transacoes.forEach(t => {
                // Cálculos
                if (t.tipo === 'entrada') {
                    totalEntradas += t.valor;
                } else {
                    totalSaidas += t.valor;
                    // Agrupa categorias
                    if (gastosPorCategoria[t.categoria]) {
                        gastosPorCategoria[t.categoria] += t.valor;
                    } else {
                        gastosPorCategoria[t.categoria] = t.valor;
                    }
                }

                // Renderiza Tabela
                const tr = document.createElement('tr');
                const corValor = t.tipo === 'entrada' ? 'text-emerald-600' : 'text-red-600';
                const iconeStatus = t.pago ? '<i class="fas fa-check-circle text-emerald-500 text-xl cursor-pointer" title="Pago"></i>' : '<i class="far fa-circle text-gray-400 text-xl cursor-pointer" title="Pendente"></i>';
                
                tr.innerHTML = `
                    <td class="px-6 py-4 whitespace-nowrap text-center" onclick="alternarStatusPagamento(${t.id})">${iconeStatus}</td>
                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-900 font-medium">${t.nome}</td>
                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500"><span class="bg-gray-100 px-2 py-1 rounded text-xs">${t.categoria}</span></td>
                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">${formatarData(t.dataVencimento)}</td>
                    <td class="px-6 py-4 whitespace-nowrap text-sm font-bold ${corValor}">${formatarMoeda(t.valor)}</td>
                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-500">
                        <button onclick="excluirTransacao(${t.id})" class="text-red-500 hover:text-red-700"><i class="fas fa-trash"></i></button>
                    </td>
                `;
                tbody.appendChild(tr);
            });

            // Atualiza Dashboard
            document.getElementById('totalEntradas').innerText = formatarMoeda(totalEntradas);
            document.getElementById('totalSaidas').innerText = formatarMoeda(totalSaidas);
            document.getElementById('saldoAtual').innerText = formatarMoeda(totalEntradas - totalSaidas);
            
            // Muda a cor do saldo se ficar negativo
            if (totalEntradas - totalSaidas < 0) {
                document.getElementById('saldoAtual').className = "text-3xl font-bold text-red-600 mt-2";
            } else {
                document.getElementById('saldoAtual').className = "text-3xl font-bold text-blue-600 mt-2";
            }

            // Atualiza Resumo por Categoria
            const divCategorias = document.getElementById('resumoCategorias');
            divCategorias.innerHTML = '';
            
            // Ordena categorias pelo valor gasto (maior pro menor)
            const categoriasOrdenadas = Object.entries(gastosPorCategoria).sort((a, b) => b[1] - a[1]);

            if (categoriasOrdenadas.length === 0) {
                divCategorias.innerHTML = '<p class="text-sm text-gray-500 col-span-2">Nenhum gasto registrado ainda.</p>';
            } else {
                categoriasOrdenadas.forEach(([cat, valor]) => {
                    let porcentagem = ((valor / totalSaidas) * 100).toFixed(1);
                    divCategorias.innerHTML += `
                        <div class="border rounded p-3 flex justify-between items-center bg-gray-50">
                            <div>
                                <p class="text-sm font-bold text-gray-700">${cat}</p>
                                <p class="text-xs text-gray-500">${porcentagem}% do total</p>
                            </div>
                            <span class="text-sm font-bold text-red-500">${formatarMoeda(valor)}</span>
                        </div>
                    `;
                });
            }
        }

        // Inicializa a interface com os dados de exemplo
        atualizarInterface();
    </script>
